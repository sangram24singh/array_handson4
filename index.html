{

const findEvenOdd = (N, Arr) => 
{
    let evensm = 0;
    let oddsm = 0;
    for (let i = 0;i < N;i++)
    {
      if (Arr[i]%2===0)
      {
        evensm+=Arr[i];
      }
      else {
        oddsm+=Arr[i];
      }
    }
    let ans = [evensm, oddsm];
    return ans;
};

Min and Max

 function arrayMin(arr) {
    min=arr[0]
    for (let i = 0; i < arr.length; i++)
    {
      if (arr[i] < min){
        min=arr[i]
      }
    }
    return min;
}

Birthday Game

function Birthday_Game(arr,D ,M) {
 let N=arr.length;
  let count=0;
  for (let i = 0; i < N-M+1; i++)
  {
    let sum =0;
    for (let j = i; j < M+i; j++){
      sum+=arr[j];
    }
    if(sum===D)
    count++
  }
  return count;
}
